# рейсы идущие из A в B, время отправления (time_from, time_to)

select a.route_id
from route_point a, route_point b
where
    a.route_id = b.route_id and
    a.station_id = A and
    b.station_id = B and
    a.departure_time < b.arrival_time and
    a.departure_time between time_from and time_to
